# convert a int List into str List
a = [1,5,10]
type(a)

b = str(a)
print(b)
print(type(b))

c = [str(i) for i in a]
print(c)
print(type(c))

# convert a int tuple into str tuple
e = [(2,3,4),(23,42,54)]
f = [tuple([str(i) for i in tp]) for tp in e]
print(e)
print(f)

# How to sort a Python dict by value
